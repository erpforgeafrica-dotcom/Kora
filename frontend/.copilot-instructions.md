# Frontend Development Instructions

## Architecture

**Framework**: React 18 + Vite + TypeScript  
**Styling**: CSS Variables (theme system)  
**State**: Zustand + Context API  
**Routing**: React Router v6  
**Auth**: Clerk

## CRUD Page Pattern

### List Page
```tsx
export function ClientsListPage() {
  const [clients, setClients] = useState([]);
  
  useEffect(() => {
    fetch("/api/clients", {
      headers: { "x-org-id": organizationId }
    })
      .then(res => res.json())
      .then(data => setClients(data.clients));
  }, []);
  
  return (
    <div style={{ padding: 24 }}>
      <div style={{ display: "flex", justifyContent: "space-between", marginBottom: 24 }}>
        <h1>Clients</h1>
        <button onClick={() => navigate("/app/clients/create")}>
          Create Client
        </button>
      </div>
      <table>
        {clients.map(client => (
          <tr key={client.id}>
            <td>{client.name}</td>
            <td>
              <button onClick={() => navigate(`/app/clients/${client.id}/edit`)}>
                Edit
              </button>
            </td>
          </tr>
        ))}
      </table>
    </div>
  );
}
```

### Create Page
```tsx
export function CreateClientPage() {
  const navigate = useNavigate();
  const [formData, setFormData] = useState({ name: "", email: "" });
  
  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    const res = await fetch("/api/clients", {
      method: "POST",
      headers: { 
        "Content-Type": "application/json",
        "x-org-id": organizationId 
      },
      body: JSON.stringify(formData)
    });
    if (res.ok) navigate("/app/clients");
  };
  
  return (
    <form onSubmit={handleSubmit}>
      {/* Form fields */}
    </form>
  );
}
```

## State Management

### Zustand Store
```typescript
// stores/useAppStore.ts
import { create } from "zustand";

interface AppState {
  organizationId: string | null;
  setOrganizationId: (id: string) => void;
}

export const useAppStore = create<AppState>((set) => ({
  organizationId: null,
  setOrganizationId: (id) => set({ organizationId: id })
}));
```

### Context API
```tsx
// contexts/ThemeContext.tsx
export const ThemeContext = createContext<{
  themeId: ThemeId;
  setTheme: (id: ThemeId) => void;
}>({ themeId: "dark", setTheme: () => {} });

export function ThemeProvider({ children }) {
  const [themeId, setThemeId] = useState<ThemeId>("dark");
  return (
    <ThemeContext.Provider value={{ themeId, setTheme: setThemeId }}>
      {children}
    </ThemeContext.Provider>
  );
}
```

### Error Handling
```typescript
const [error, setError] = useState<string | null>(null);

try {
  const res = await fetch("/api/bookings", { method: "POST", body: JSON.stringify(data) });
  if (!res.ok) throw new Error("Failed to create booking");
  const booking = await res.json();
} catch (err) {
  setError(err instanceof Error ? err.message : "Unknown error");
}
```

## Anti-Patterns

### ❌ DON'T
```tsx
// Hardcoded colors
<div style={{ background: "#00e5c8" }}>

// Inline styles exceeding 50 lines
<div style={{ /* 100 lines of styles */ }}>

// Duplicate components
// Creating custom Button when Team B's exists

// Missing error handling
const data = await fetch("/api/bookings").then(r => r.json());

// No loading states
return <div>{data.map(...)}</div>;
```

### ✅ DO
```tsx
// Use theme variables
<div style={{ background: "var(--color-accent)" }}>

// Extract to component
<StyledCard>...</StyledCard>

// Use Team B components
import { Button } from "../components/ui/Button";

// Handle errors
try {
  const data = await fetch("/api/bookings").then(r => r.json());
} catch (err) {
  setError(err.message);
}

// Show loading states
if (loading) return <div>Loading...</div>;
```
